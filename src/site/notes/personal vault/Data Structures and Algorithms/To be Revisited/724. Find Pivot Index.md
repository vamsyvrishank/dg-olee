Solution:

1. SC = O(n) & TC = O(n)

```java
class Solution {
    public int pivotIndex(int[] nums) {
	    // Include a placeholder at index 0
        int[] prefixSum = new int[nums.length + 1];

        for (int i = 1; i <= nums.length; i++) {
            prefixSum[i] = prefixSum[i - 1] + nums[i - 1];
        }

        for (int i = 1; i < prefixSum.length; i++) {  // Start from index 1
	        // Compare left and right sums correctly
            if (prefixSum[i - 1] == prefixSum[prefixSum.length - 1] - prefixSum[i]) 
            {
	            return i - 1;  // Adjust for the placeholder
            }
        }

        return -1;
    }
}
```

 2. SC = O(1) & TC = O(n)

```java
class Solution {
    public int pivotIndex(int[] nums) {
        int leftSum = 0;
        int sum = 0;
        for(int num: nums){
            sum += num;
        }

        for(int i = 0; i < nums.length; i++){
            if(leftSum == sum - leftSum - nums[i]){
                return i;
            }
            leftSum += nums[i];
        }

        return -1;
    }
}
```
