Given an integer array `nums` of **unique** elements, return _all possible_ 

_subsets_

 _(the power set)_.

The solution set **must not** contain duplicate subsets. Return the solution in **any order**.

**Example 1:**

**Input:** nums = [1,2,3]
**Output:** [[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3\|],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]

**Example 2:**

**Input:** nums = [0]
**Output:** [[],[0\|],[0]]

**Constraints:**

- `1 <= nums.length <= 10`
- `-10 <= nums[i] <= 10`
- All the numbers of `nums` are **unique**.


```cpp
class Solution {
public:
    void backtrack(int index ,vector<int> &nums,vector<int> poss_ans , vector<vector<int>> &ans){
        if(index==nums.size()){
            ans.push_back(poss_ans);
            return;
        }
        
        //pick it;
        poss_ans.push_back(nums[index]);
        backtrack(index+1,nums,poss_ans,ans);
        poss_ans.pop_back();

        //dont pick;
        backtrack(index+1,nums,poss_ans,ans);
    }
    vector<vector<int>> subsets(vector<int>& nums) {

        vector<vector<int>> ans;
        vector<int>poss_ans;
        int index = 0;
        backtrack(index,nums,poss_ans,ans);
        return ans;

        
    }
};
```
