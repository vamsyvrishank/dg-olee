Given an integer array `nums` that may contain duplicates, return _all possible_ 

_subsets_

 _(the power set)_.

The solution set **must not** contain duplicate subsets. Return the solution in **any order**.

**Example 1:**

**Input:** nums = [1,2,2]
**Output:** [[],[1],[1,2],[1,2,2],[2],[2,2\|],[1],[1,2],[1,2,2],[2],[2,2]]

**Example 2:**

**Input:** nums = [0]
**Output:** [[],[0\|],[0]]

**Constraints:**

- `1 <= nums.length <= 10`
- `-10 <= nums[i] <= 10`


```cpp

class Solution {
public:
    void backtrack(int index,vector<int> &nums, vector<int> poss_ans, vector<vector<int>> &ans){

        if(index == nums.size()){
            ans.push_back(poss_ans);
            return;
        }
        
        // pick it
        poss_ans.push_back(nums[index]);
        backtrack(index+1,nums,poss_ans,ans);
        poss_ans.pop_back();
        
		// after you have picked the elements , when not picking ,  skip the same elements to avoid duplicate set generation.
		//very cool approach
        while(index+1 <= nums.size()-1 and nums[index] == nums[index+1]){
            index++;
        }

        //dont pick
        backtrack(index+1,nums,poss_ans,ans);
        return;
    }
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        
        vector<vector<int>> ans;
        vector<int> poss_ans;
        int index = 0;

        sort(nums.begin(),nums.end());
        backtrack(index,nums,poss_ans,ans);

        return ans;
    }
};
```
